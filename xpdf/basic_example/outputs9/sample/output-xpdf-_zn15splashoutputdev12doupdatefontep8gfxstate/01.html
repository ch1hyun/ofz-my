void process(PDFDoc* doc, FuzzedDataProvider* data_provider) {
  if (!doc->okToPrint()) {
    return;
  }
  const int pages = doc->getNumPages();
  for (int p = 1; p <= pages; ++p) {
    Catalog* catalog = doc->getCatalog();
    Page* page = catalog->getPage(p);
    if(!page) {
      continue;
    }
    Object obj;
    page->getContents(&obj);
    try {
      auto *gfx_state = new GfxState(72, 72, page->getCropBox(), 0, gFalse);
      gfx_state->setFont(GfxFont::makeFont(doc->getXRef(), "Helvetica",
          obj.getRef(), NULL));
      
      SplashColor paperColor;
      paperColor[0] = paperColor[1] = paperColor[2] = 255; // white
      
      std::unique_ptr<SplashOutputDev> splash_output_dev = std::make_unique<
              SplashOutputDev>(splashModeRGB8, 4, gFalse, paperColor);
      
      splash_output_dev->startPage(0, gfx_state);
      delete gfx_state;
    } catch(...) {
    }
  }
}
}
