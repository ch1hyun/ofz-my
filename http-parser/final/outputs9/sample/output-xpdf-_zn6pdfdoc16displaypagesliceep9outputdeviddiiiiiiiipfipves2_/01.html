  double vdpi = fdp.ConsumeFloatingPoint<double>();
  int rotate = fdp.ConsumeIntegral<int>();
  bool useMediaBox = fdp.ConsumeBool();
  bool crop = fdp.ConsumeBool();
  bool printing = fdp.ConsumeBool();
  std::vector<char> payload = fdp.ConsumeRemainingBytes<char>();

  Object xpdf_obj;
  xpdf_obj.initNull();
  BaseStream *stream = new MemStream(payload.data(), 0, payload.size(), &xpdf_obj);

  GlobalParams *globalParams = new GlobalParams(NULL);
  globalParams->setErrQuiet(1);
  globalParams->setupBaseFonts(NULL);
  char yes[4] = "yes";
  globalParams->setEnableFreeType(yes);
  globalParams->setErrQuiet(1);

  PDFDoc doc(stream);

  if (doc.isOk() == gTrue) {
    GString *metadata;
    if ((metadata = doc.readMetadata())) {
      (void)metadata->getCString();
    }
    delete metadata;

    // Creating splash output device
    SplashColor paperColor = {0xff, 0xff, 0xff};
    SplashOutputDev *splashOut = new SplashOutputDev(splashModeRGB8, 1, gFalse, paperColor);
    splashOut->setNoComposite(gTrue);
    splashOut->startDoc(doc.getXRef());

    // Define values to call displayPageSlice
    int sliceX = 0;
    int sliceY = 0;
    int sliceW = 0;
    int sliceH = 0;

    // Getting pages
    for (size_t i = 1; i <= doc.getNumPages(); ++i) {
      doc.displayPageSlice(splashOut, i, hdpi, vdpi, rotate, useMediaBox, crop, printing,
                           sliceX, sliceY, sliceW, sliceH, NULL, NULL);
      }

    delete splashOut;
  }

  delete globalParams;

  return 0;
}
