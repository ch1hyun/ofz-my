  GBool invisible = fdp.ConsumeBool();
  GBool freeData = fdp.ConsumeBool();
  GBool adjustXY = fdp.ConsumeBool();

  Object obj;

  int rotate = 0;
  int pageNum = 1;

  GfxState *state = new GfxState(72, 72, &mediaBox, rotate, gFalse);
  SplashColor paperColor = { 255, 255, 255 }; // white

  splashOut->startPage(pageNum, state);
  splashOut->drawChar(state, x, y, dx, dy, originX, originY, code, nBytes, u, nBytes, invisible, freeData, adjustXY);

  delete splashOut;
  delete state;
  delete doc;
  delete globalParams; // Don't forget to delete the GlobalParams object

  return 0;
}
